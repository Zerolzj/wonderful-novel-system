        }

    }
    
    findEntangledPairs(universes) {
        const pairs = [];
        
        for (let i = 0; i < universes.length; i++) {
            for (let j = i + 1; j < universes.length; j++) {
                if (this.isEntangled(universes[i], universes[j])) {
                    pairs.push([universes[i], universes[j]]);
                }
            }
        }
        
        return pairs;
    }

        const quantum1 = uni1.quantumSignature;
        const quantum2 = uni2.quantumSignature;
        
        return this.calculateEntanglement(quantum1, quantum2) > 0.8;
