    }

            "sadness": ["#4169E1", "#6495ED", "#778899", "#B0C4DE"],
            "anger": ["#DC143C", "#FF4500", "#FF6347", "#CD5C5C"],
            "love": ["#FF1493", "#FF69B4", "#FFB6C1", "#FFC0CB"],
            "hope": ["#90EE90", "#98FB98", "#00FF7F", "#00FA9A"]
        };
        return colorPalette[emotion] || colorPalette["joy"];
    }
    
    paint(feeling_intensity) {
        let x = Math.random() * this.canvas.width;
        let y = Math.random() * this.canvas.height;
        let color = this.colors[Math.floor(Math.random() * this.colors.length)];
        let size = feeling_intensity * 10;
        
        this.canvas.drawCircle(x, y, size, color);

            }, i * 100);
        }

    compose(emotional_journey) {
        // 将情感旅程转化为艺术作品

            this.paint(moment.intensity);
            this.sleep(moment.duration);
        }
    }
}
