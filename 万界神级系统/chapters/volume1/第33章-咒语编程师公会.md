```
def heal(target, power=medium):
    """
    治疗咒语
    target: 治疗目标
    power: 治疗强度 (low/medium/high)
    """
    mana_cost = {
        'low': 20,
        'medium': 50,
        'high': 100
    }

        'medium': 300,

```

```python
class FireSpell:
    def __init__(self, caster):
        self.caster = caster
        self.base_power = caster.magic_power * 0.8
    
    def fireball(self, target):
        return self._cast_skill('fireball', target, self.base_power * 1.5)
    
    def fire_wall(self, duration=10):
        return self._cast_skill('fire_wall', None, self.base_power * 0.5, duration)
    
    def _cast_skill(self, skill_name, target, power, *args):
        if self.caster.mana >= power:
            execute_magic(skill_name, target, power, *args)
            self.caster.mana -= power
            return True
        return False
```

```python
def scout_area(radius=100):
    """侦察周围区域，标记所有敌人"""
    enemies = []
    for direction in range(0, 360, 30):
        result = magic_scan(direction, radius)
        if result.life_form and result.hostile:
            enemies.append({
                'position': result.position,
                'type': result.creature_type,
                'threat_level': result.power_level
            })
    return enemies
```

```python
class VillageDefense:
    def __init__(self):
        self.active_defenses = []
        self.mana_pool = 500
    
    def add_barrier(self, position, strength=medium):
        """在指定位置添加魔法屏障"""
        cost = strength * 20
        if self.mana_pool >= cost:
            barrier = create_barrier(position, strength)
            self.active_defenses.append(barrier)
            self.mana_pool -= cost
            return True
        return False
    
    def auto_defend(self, enemies):
        """自动防御系统"""
        for enemy in enemies:
            if enemy.threat_level > 5:
                self.add_barrier(enemy.position, 'high')
            elif enemy.threat_level > 2:
                self.add_barrier(enemy.position, 'medium')
```
