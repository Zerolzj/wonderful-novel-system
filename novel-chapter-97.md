```python
class Universe:
    def __init__(self):
        self.observers = {
            'internal': set(),    # 内部观察者
            'external': set(),    # 外部观察者
            'meta': set()         # 元观察者
        }
        self.life_exists = False
        self.stability_factor = 1.0
        self.recursion_depth = 0
    
    def check_life(self):
        """检查是否存在生命"""
        self.life_exists = random.choice([True, False])
        return self.life_exists
    
    def add_observer(self, observer, observer_type='external'):
        """添加观察者，支持层级管理"""
        if observer_type not in self.observers:
            raise ValueError(f"Invalid observer type: {observer_type}")
        
        self.observers[observer_type].add(observer)
        
        # 修复后的逻辑：只有外部观察者可以触发内部观察
        if observer_type == 'external' and self.check_life():
            if self.recursion_depth < 1:  # 限制递归深度
                self.recursion_depth += 1
                self.add_observer(self, 'meta')  # 宇宙作为元观察者
                self.recursion_depth -= 1
    
    def calculate_stability(self):
        """改进的稳定性计算"""
        internal_count = len(self.observers['internal'])
        external_count = len(self.observers['external'])
        meta_count = len(self.observers['meta'])
        
        # 元观察者不会影响稳定性
        effective_observers = internal_count + external_count
        
        # 使用对数函数代替线性函数，减少观察者数量对稳定性的影响
        self.stability_factor = 1.0 / (1.0 + math.log(1 + effective_observers) * 0.1)
        
        # 元观察者提供稳定性加成
        if meta_count > 0:
            self.stability_factor *= 1.2
            
        return min(self.stability_factor, 1.0)  # 确保不超过1.0
```
