def spiral_formation(team_members, enemy_data):
    # 初始化螺旋阵法参数
    angle_step = 72  # 五人阵法，每人72度
    radius = 10      # 阵法半径
    rotation_speed = 1.5  # 旋转速度倍率
    
    # 计算每个人的位置和攻击角度
    positions = []
    for i, member in enumerate(team_members):
        angle = i * angle_step
        x = radius * math.cos(angle)
        y = radius * math.sin(angle)
        attack_angle = angle + 90  # 攻击角度垂直于位置
        positions.append({
            'member': member,
            'position': (x, y),
            'attack_angle': attack_angle
        })

        radius *= 0.8  # 缩小阵法，增强防御

    return positions
